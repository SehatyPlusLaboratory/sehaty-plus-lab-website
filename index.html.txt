    <script>
        document.addEventListener('DOMContentLoaded', function () {
            const mobileMenuButton = document.getElementById('mobile-menu-button');
            const mobileMenu = document.getElementById('mobile-menu');
            // Convert NodeLists to arrays and filter out any potential nulls (though querySelectorAll shouldn't return them)
            const langButtons = Array.from(document.querySelectorAll('.language-button')).filter(Boolean);
            const langContents = Array.from(document.querySelectorAll('.lang-content')).filter(Boolean);
            const navLinks = Array.from(document.querySelectorAll('.nav-link')).filter(Boolean);
            const mobileNavLinks = Array.from(document.querySelectorAll('.mobile-nav-link')).filter(Boolean);

            // Function to set the active language
            function setLanguage(lang) {
                try {
                    // Hide all content divs
                    langContents.forEach(content => {
                        content.classList.add('hidden');
                    });

                    // Remove 'active' class from all language buttons
                    langButtons.forEach(button => {
                        button.classList.remove('active');
                    });

                    // Show selected language content div
                    const selectedLangContent = document.getElementById(`${lang}-content`);
                    if (selectedLangContent) {
                        selectedLangContent.classList.remove('hidden');
                    } else {
                        console.error(`Error: Language content div '${lang}-content' not found.`);
                    }

                    // Set 'active' class for the correct language buttons (desktop and mobile)
                    const desktopLangButton = document.getElementById(`lang-${lang}`);
                    if (desktopLangButton) {
                        desktopLangButton.classList.add('active');
                    } else {
                        console.error(`Error: Desktop language button 'lang-${lang}' not found.`);
                    }
                    const mobileLangButton = document.getElementById(`mobile-lang-${lang}`);
                    if (mobileLangButton) {
                        mobileLangButton.classList.add('active');
                    } else {
                        console.error(`Error: Mobile language button 'mobile-lang-${lang}' not found.`);
                    }

                    // Update text direction for Arabic
                    if (lang === 'ar') {
                        document.body.style.direction = 'rtl';
                    } else {
                        document.body.style.direction = 'ltr';
                    }

                    // Update navigation links text for both desktop and mobile
                    navLinks.forEach(link => {
                        if (link.dataset[lang]) { // Check if the specific language data attribute exists
                            link.textContent = link.dataset[lang];
                        } else {
                            console.warn(`Warning: Navigation link dataset for '${lang}' not found or incomplete:`, link);
                        }
                    });
                    mobileNavLinks.forEach(link => {
                        if (link.dataset[lang]) { // Check if the specific language data attribute exists
                            link.textContent = link.dataset[lang];
                        } else {
                            console.warn(`Warning: Mobile navigation link dataset for '${lang}' not found or incomplete:`, link);
                        }
                    });

                    // Store selected language in local storage
                    localStorage.setItem('preferredLanguage', lang);
                } catch (e) {
                    console.error("An error occurred in setLanguage:", e);
                }
            }

            // Event listeners for language buttons
            langButtons.forEach(button => {
                button.addEventListener('click', function () {
                    const lang = this.id.includes('en') ? 'en' : 'ar';
                    setLanguage(lang);
                });
            });

            // Mobile menu toggle
            if (mobileMenuButton && mobileMenu) { // Ensure both elements exist before adding listener
                mobileMenuButton.addEventListener('click', function () {
                    mobileMenu.classList.toggle('hidden');
                });
            } else {
                console.error("Error: Mobile menu button or mobile menu element not found.");
            }

            // Close mobile menu when a link is clicked
            mobileNavLinks.forEach(link => {
                link.addEventListener('click', function () {
                    if (mobileMenu) { // Ensure mobileMenu exists before trying to hide it
                        mobileMenu.classList.add('hidden');
                    }
                });
            });

            // Load preferred language from local storage or default to English on initial page load
            const storedLang = localStorage.getItem('preferredLanguage');
            setLanguage(storedLang || 'en');
        });
    </script>
